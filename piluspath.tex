\begin{block}{Path ($A \rightarrow C$) computation}
\small
MHMC algorithm:
$$
P(x_{t+1}=x|x_t)=\min\left\{\frac{\pi(x)Q(x)}{\pi(x_t)Q(x_t)},1\right\}
$$
with:
$$
Q(x) = \exp(-d_{x}/k)
$$
and:
$$
\pi(x) = \exp(-U(x)/k')
$$
An average on 100 possible paths was computed\\
\end{block}}
